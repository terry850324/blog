    Dim obj_SmtpClient As New SmtpClient
    Dim obj_SmtpMessage As New MailMessage

    Try

        ' Mail content
        obj_SmtpMessage.From = New MailAddress(str_MailAct)
        For Each str_MailAddress As String In strlst_MailToAddress
            obj_SmtpMessage.To.Add(str_MailAddress)
        Next
        obj_SmtpMessage.Subject = str_Subject
        obj_SmtpMessage.IsBodyHtml = True
        obj_SmtpMessage.Body = str_MailBody

        ' SMTP setting
        obj_SmtpClient.Host = "CAS.sigurd.com.tw"
        obj_SmtpClient.Port = "25"
        obj_SmtpClient.Credentials = New NetworkCredential(str_MailAct, str_MailPwd)

        ' Send mail
        obj_SmtpClient.Send(obj_SmtpMessage)

    Catch ex As Exception
        ConsoleWriteLine("Sending mail failed!" & vbNewLine & ex.ToString, ConsoleColor.Red)
    End Try

End Sub
